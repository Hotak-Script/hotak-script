50 SQL hacks for querying, joins, performance, and reliable data work.

Hack 1: Select specific columns

SELECT
    id,
    name
FROM users;

Hack 2: Filter with WHERE

SELECT *
FROM users
WHERE salary > 50000;

Hack 3: Sort results

SELECT *
FROM users
ORDER BY created_at DESC;

Hack 4: Paginate rows

SELECT *
FROM users
LIMIT 20 OFFSET 40;

Hack 5: Aggregate counts

SELECT
    COUNT(*) AS total_users
FROM users;

Hack 6: Group and summarize

SELECT
    gender,
    AVG(salary)
FROM users
GROUP BY gender;

Hack 7: Inner join

SELECT
    u.name,
    a.city
FROM users u
INNER JOIN addresses a
    ON u.id = a.user_id;

Hack 8: Upsert style insert

INSERT INTO users (id, name)
VALUES (1, "Ali")
ON DUPLICATE KEY UPDATE
    name = VALUES(name);

Hack 9: Safe transaction

START TRANSACTION;
UPDATE users SET salary = salary + 1000 WHERE id = 2;
COMMIT;
