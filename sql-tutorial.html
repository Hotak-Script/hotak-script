The Ultimate SQL Tutorial

"MySQL is a database management system. A Database Management System (DBMS) is software that interacts with end users, applications, and the database itself to capture and analyze data."

By CodeWithHarry Updated: July 13, 2025 MySQL + SQL Complete

Installing MySQL

What is MySQL Workbench? A visual tool for DB architects, developers, and DBAs with data modeling, SQL development, and admin tools.

Windows/macOS: Download installer from `https://dev.mysql.com/downloads/installer/`, choose Developer Default, set root password, and install Workbench.

Ubuntu Linux:

sudo apt update
sudo apt install mysql-server
sudo mysql_secure_installation

sudo mysql
CREATE USER 'harry'@'localhost' IDENTIFIED BY 'password';
GRANT ALL PRIVILEGES ON *.* TO 'harry'@'localhost' WITH GRANT OPTION;
FLUSH PRIVILEGES;
EXIT;

mysql -u harry -p

Getting Started

A database stores related data in an organized way. In MySQL, one database contains one or more tables.

CREATE DATABASE startersql;
USE startersql;

CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  email VARCHAR(100) UNIQUE NOT NULL,
  gender ENUM('Male', 'Female', 'Other'),
  date_of_birth DATE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

Delete everything in DB:

DROP DATABASE startersql;

Working with Tables

SELECT * FROM users;
SELECT name, email FROM users;

RENAME TABLE users TO customers;
RENAME TABLE customers TO users;

ALTER TABLE users ADD COLUMN is_active BOOLEAN DEFAULT TRUE;
ALTER TABLE users DROP COLUMN is_active;
ALTER TABLE users MODIFY COLUMN name VARCHAR(150);
ALTER TABLE users MODIFY COLUMN email VARCHAR(100) FIRST;

Insert and Query Data

INSERT INTO users (name, email, gender, date_of_birth) VALUES
('Bob', 'bob@example.com', 'Male', '1990-11-23'),
('Charlie', 'charlie@example.com', 'Other', '1988-02-17');

SELECT * FROM users WHERE gender = 'Male';
SELECT * FROM users WHERE date_of_birth BETWEEN '1990-01-01' AND '2000-12-31';
SELECT * FROM users WHERE name LIKE 'A%';
SELECT * FROM users ORDER BY created_at DESC LIMIT 10;

UPDATE and DELETE

UPDATE users
SET name = 'Alicia'
WHERE id = 1;

UPDATE users
SET salary = salary + 10000
WHERE salary < 60000;

DELETE FROM users WHERE id = 3;
DELETE FROM users WHERE salary IS NULL;
DROP TABLE users;

Tip: Always run matching SELECT before update/delete on production data.

Constraints

-- UNIQUE
ALTER TABLE users ADD CONSTRAINT unique_email UNIQUE (email);

-- NOT NULL
ALTER TABLE users MODIFY COLUMN name VARCHAR(100) NOT NULL;

-- CHECK
ALTER TABLE users
ADD CONSTRAINT chk_dob CHECK (date_of_birth > '2000-01-01');

-- DEFAULT
ALTER TABLE users ALTER COLUMN is_active SET DEFAULT TRUE;

-- PRIMARY KEY + AUTO_INCREMENT
CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100)
);

Functions

SELECT COUNT(*) FROM users;
SELECT MIN(salary), MAX(salary) FROM users;
SELECT AVG(salary) FROM users;
SELECT gender, AVG(salary) FROM users GROUP BY gender;

SELECT name, LENGTH(name) FROM users;
SELECT CONCAT(name, ' <', email, '>') FROM users;

SELECT NOW();
SELECT name, TIMESTAMPDIFF(YEAR, date_of_birth, CURDATE()) AS age FROM users;

SELECT name, IF(gender='Female','Yes','No') AS is_female FROM users;

Transactions and AutoCommit

SET autocommit = 0;
UPDATE users SET salary = 80000 WHERE id = 5;
COMMIT;
-- or
ROLLBACK;
SET autocommit = 1;

Foreign Keys and JOINs

CREATE TABLE addresses (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT,
  city VARCHAR(100),
  CONSTRAINT fk_user FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

SELECT u.name, a.city
FROM users u
INNER JOIN addresses a ON u.id = a.user_id;

SELECT u.name, a.city
FROM users u
LEFT JOIN addresses a ON u.id = a.user_id;

UNION, Views, Indexes, Subqueries

SELECT name FROM users
UNION
SELECT name FROM admin_users;

CREATE VIEW high_salary_users AS
SELECT id, name, salary FROM users WHERE salary > 70000;

CREATE INDEX idx_email ON users(email);
CREATE INDEX idx_gender_salary ON users(gender, salary);

SELECT id, name, salary
FROM users
WHERE salary > (SELECT AVG(salary) FROM users);

GROUP BY, HAVING, Procedures, Triggers

SELECT gender, AVG(salary) AS avg_salary
FROM users
GROUP BY gender
HAVING AVG(salary) > 75000;

DELIMITER $$
CREATE PROCEDURE AddUser(
  IN p_name VARCHAR(100),
  IN p_email VARCHAR(100),
  IN p_gender ENUM('Male','Female','Other'),
  IN p_dob DATE,
  IN p_salary INT
)
BEGIN
  INSERT INTO users(name,email,gender,date_of_birth,salary)
  VALUES(p_name,p_email,p_gender,p_dob,p_salary);
END$$
DELIMITER ;

DELIMITER $$
CREATE TRIGGER after_user_insert
AFTER INSERT ON users
FOR EACH ROW
BEGIN
  INSERT INTO user_log(user_id,name) VALUES(NEW.id, NEW.name);
END$$
DELIMITER ;
